# The Flutter tooling requires that developers have CMake 3.10 or later
# installed. You should not increase this version, as doing so will cause
# the plugin to fail to compile for some customers of the plugin.
cmake_minimum_required(VERSION 3.10)

project(anisthesia_library VERSION 0.0.1 LANGUAGES C)

add_library(anisthesia SHARED
  "anisthesia.cpp"
)

set_target_properties(anisthesia PROPERTIES
  LINKER_LANGUAGE CXX
  PUBLIC_HEADER anisthesia.hpp
  OUTPUT_NAME "anisthesia"
)

target_compile_definitions(anisthesia PUBLIC DART_SHARED_LIB)
